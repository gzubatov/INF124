CREATE table `products_in_orders` (`oid` int(11) NOT NULL,
`pid` int(11) NOT NULL,
`quantity` int(11) NOT NUll,
PRIMARY KEY(`oid`, `pid`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `categories` (`cid` int(11) NOT NULL AUTO_INCREMENT,
  `category` text NOT NULL,
  PRIMARY KEY (`cid`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8mb4;	

CREATE TABLE `orders` (`oid` int(11) NOT NULL AUTO_INCREMENT,
 `pid` int(11) NOT NULL,
 `quantity` int(11) NOT NULL,
 `first_name` text NOT NULL,
 `last_name` text NOT NULL,
 `phone_number` text NOT NULL,
 `shipping_address` text NOT NULL,
 `zip_code` bigint(20) NOT NULL,
 `shipping_method` text NOT NULL,
 `credit_card` bigint(20) unsigned NOT NULL,
 `expiration_month` int(11) NOT NULL,
 `expiration_year` int(11) NOT NULL,
 `security_code` int(11) NOT NULL,
 `price_total` double(25,2) NOT NULL,
 PRIMARY KEY (`oid`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4

CREATE TABLE `products` (
 `pid` int(11) NOT NULL AUTO_INCREMENT,
 `name` text NOT NULL,
 `price` decimal(5,2) NOT NULL,
 `description` text NOT NULL,
 `image` text NOT NULL,
 `details` text NOT NULL,
 PRIMARY KEY (`pid`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8mb4

CREATE TABLE `product_has_category` (
 `pid` int(11) NOT NULL,
 `cid` int(11) NOT NULL,
 PRIMARY KEY (`pid`,`cid`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4

CREATE TABLE `tax_rates` (
 `state` text NOT NULL,
 `zip` bigint(20) NOT NULL,
 `tax_region_name` text NOT NULL,
 `combined_rate` double NOT NULL,
 PRIMARY KEY (`zip`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4

CREATE TABLE `zip_codes` (
 `zip` bigint(20) NOT NULL,
 `state` text NOT NULL,
 `city` text NOT NULL,
 PRIMARY KEY (`zip`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4
